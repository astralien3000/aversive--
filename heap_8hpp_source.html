150  void percolateUp(heap_t index) {
151  Node cur(this, index);
152  if(cur.isValid()) {
153  while(cur.hasParent() && cur.parent().element() < cur.element()) {
154  cur.switchElements(cur.parent());
155  cur = cur.parent();
156  }

162  void percolateDown(heap_t index) {
163  Node cur(this, index);
164  if(cur.isValid()) {
165  // While there is one child greater than current
166  while((cur.hasFirstChild() &&
167  cur.element() < cur.firstChild().element()) ||
168  (cur.hasSecondChild() &&
169  cur.element() < cur.secondChild().element())) {
170  // If there is only one child
171  // The first child must be filled first !
172  if(!cur.hasFirstChild()) {
173  cur.switchElements(cur.secondChild());
174  cur = cur.secondChild();
175  }
176  else if(!cur.hasSecondChild()) {
177  cur.switchElements(cur.firstChild());
178  cur = cur.firstChild();
179  }
180  // If there is two children
181  else {
182  // Switch with the max
183  if(cur.firstChild().element() < cur.secondChild().element()) {
184  cur.switchElements(cur.secondChild());
185  cur = cur.secondChild();
186  }
187  else {
188  cur.switchElements(cur.firstChild());
189  cur = cur.firstChild();
190  }
191  }
192  }
193  }
